# =================================================== # Tarea: Calculadora con Menú # Participante A: Condicionales (menú y operaciones) # Participante B: Loops (repetición y validación) # =================================================== def mostrar_menu(): print("\n--- CALCULADORA ---") print("1. Sumar") print("2. Restar") print("3. Multiplicar") print("4. Dividir") print("5. Salir") # --- PARTE DE A: condicionales para elegir la operación --- def realizar_operacion(opcion, num1, num2): if opcion == "1": resultado = num1 + num2 print(f"Resultado: {num1} + {num2} = {resultado}") elif opcion == "2": resultado = num1 - num2 print(f"Resultado: {num1} - {num2} = {resultado}") elif opcion == "3": resultado = num1 * num2 print(f"Resultado: {num1} * {num2} = {resultado}") elif opcion == "4": if num2 == 0: print("Error: no se puede dividir entre cero.") else: resultado = num1 / num2 print(f"Resultado: {num1} / {num2} = {resultado}") else: print("Opción inválida.") def main(): while True: mostrar_menu() opcion = input("Elige una opción (1-5): ") if opcion == "5": print("Programa finalizado. ¡Adiós!") break if opcion not in ["1", "2", "3", "4"]: print("Opción no válida, intenta de nuevo.") continue # Validación de números con loop while True: try: num1 = float(input("Ingresa el primer número: ")) num2 = float(input("Ingresa el segundo número: ")) break except ValueError: print("Error: debes ingresar solo números.") realizar_operacion(opcion, num1, num2) main()
